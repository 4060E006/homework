#include <bits/stdc++.h>
using namespace std;

int main()
{

string str1("first string");  
string str2(str1);
string str3(5 , '#');
string str4(str1, 6, 3);
string str5(str2.begin(), str2.begin() + 4);


cout << str1 << endl;  // 列印first string 
cout << str2 << endl;  // 列印str1的值 
cout << str3 << endl;  // 印出5個# 
cout << str4 << endl;  // 取str1的值 印出第6位後3個字 
cout << str5 << endl;  // 取str2的初始值 印出起始到後4個字 


return 0;
}
