## 檢查檔案是否存在==>使用fail()

```
#include <iostream>
#include <fstream>
using namespace std;

int main()
{
  // Open a file
  ifstream input("d:///scores.txt");
  
  // 檢查檔案是否存在==>使用fail()
  if (input.fail())
  {
    cout << "找不到這個檔案" << endl;
   
    return 0;
  }

  double sum = 0;
  double number;
  
  //參看課本不同做法與錯誤分析
  while (input >> number) // Read data to the end of file
  {
    cout << number << " "; // Display data
    sum += number;
  }
  //
  input.close();

  cout << "\nTotal is " << sum << endl;

  return 0;
}
```

## 輸入想找的檔案名稱然後搜尋

```
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main()
{
  string filename;
  cout << "Enter a file name: ";
  cin >> filename;

  ifstream input(filename.c_str());
  
  if (input.fail())
    cout << filename << " 不存在" << endl;
  else
    cout << filename << " 存在" << endl;

  return 0;
}
```
