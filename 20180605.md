
## 把資料寫到檔案

```
#include <iostream>
#include <fstream>
using namespace std;

int main()
{
  ofstream output;

  // Create a file
  output.open("d:///4060E///scores.txt");

  // Write two lines
  output << "可" << " " << "以" << " " << "打" 
    << " " << "中文" << endl;
    
  output << "GH" << " " << "K" << " " << "Jones" 
    << " " << 85 << endl;
    
  output << "Eric" << " " << "K" << " " << "Jones" 
    << " " << 85 << endl; 
    
  output.close();

  cout << "OK" << endl;

  return 0;
}
```
