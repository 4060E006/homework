#include <iostream>
using namespace std;

int main()
{

string s1("Welcome");
s1.append(" to CPP"); 
cout << s1 << endl; 

string s2("Welcome");
s2.append(" to C and Cpp", 8, 4); //S字串第8個開始顯示後面4位 
cout << s2 << endl;

string s3("Welcome");
s3.append(" to C and Cpp", 5); // S字串第1個開始顯示後面5位 
cout << s3 << endl; 

string s4("Welcome"); 
s4.append(14, 'F'); 
cout << s4 << endl;

return 0;
}
